9use crate::{BillData, FontLibrary, FontStyle, Language, SlipPart};

20
21pub struct ReceiptLayout<'a>{
22    layout: BillLayout<'a>,
23    blocks: Vec<Box<dyn LayoutBlock>>
24}
25
26impl<'a> ReceiptLayout<'a> {
27    pub fn new(
28        bill_data: &'a BillData,
29        language: Language,
30        label_font_size: Pt,
31        text_font_size: Pt,
32        line_spacing: Mm,
33        extra_spacing: Mm,
34        title_ascender: Mm,
35        label_ascender: Mm,
36        text_ascender: Mm,
37    ) -> Self {
38            let layout = BillLayout {
39            bill_data,
40            config: BillLayoutConfig {
41                has_acceptance_point: true,
42                max_height: RECEIPT_MAX_HEIGHT,
43                debtor_box_height: DEBTOR_BOX_HEIGHT_RC,
44                amount_section_top: AMOUNT_SECTION_TOP,
45            },
46                language,
47                label_font_size,
48                text_font_size,
49                line_spacing,
50                extra_spacing,
51                title_ascender,
52                label_ascender,
53                text_ascender,
54            };
55        Self{
56            layout,
57            blocks: vec![
58                Box::new(TitleBlock { label: LabelKey::Receipt }),
59                Box::new(InformationBlock {part: SlipPart::Receipt, offset: Mm(0f32), payable_box_width: DEBTOR_BOX_WIDTH_RC, payable_box_height: DEBTOR_BOX_HEIGHT_RC}),
60                Box::new(SpacerBlock { min_height: Mm(260f32) }),
61                Box::new(AmountBlock{part: SlipPart::Receipt, amount_box_width: AMOUNT_BOX_WIDTH_RC, amount_box_height: AMOUNT_BOX_HEIGHT_RC}),
62            ]
63        }
64    }
65
66    pub fn layout_acceptance_point(&mut self, ops: &mut Vec<DrawOp>, fonts: &FontLibrary) {
67        let y = A4_PAGE_HEIGHT - ACCEPTANCE_POINT_SECTION_TOP + fonts.ascender_mm(FontStyle::Bold, RC_LABEL_PREF_FONT_SIZE);
68        let label_text = crate::language::label(LabelKey::AcceptancePoint, self.layout.language)
69            .unwrap_or("Acceptance point");
70        let text_width_mm = fonts.bold.measure(label_text, 6.0);
71
72        ops.push(DrawOp::Text {
73            text: label_text.to_string(),
74            at: Baseline {
75                x: Mm(RECEIPT_WIDTH.0 - MARGIN.0 - text_width_mm),
76                y: LayoutY(y),
77            },
78            size: self.layout.label_font_size,
79            bold: true,
80        });
81    }
82
83    pub fn render(&mut self, ops: &mut Vec<DrawOp>, fonts: &FontLibrary) {
84
85        self.layout.compute_spacing();
86
87        let mut main_cursor = ColumnCursor::new(
88            MARGIN,
89            A4_PAGE_HEIGHT - Mm(100f32),
90        );
91
92        // Not possible to render in a loop
93        // Spacer depends on cursor position.
94        for block in &self.blocks {
95            block.render(&mut self.layout, ops, &mut main_cursor)
96        }
97
98        self.layout_acceptance_point(ops, fonts);
99    }
100}
