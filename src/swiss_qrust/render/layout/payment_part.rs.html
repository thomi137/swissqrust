19pub struct PaymentPartLayout<'a> {
20    layout: BillLayout<'a>,
21    blocks: Vec<Box<dyn LayoutBlock>>
22}
23
24
25impl<'a> PaymentPartLayout<'a> {
26    pub fn new(
27        bill_data: &'a BillData,
28        language: Language,
29        label_font_size: Pt,
30        text_font_size: Pt,
31        line_spacing: Mm,
32        extra_spacing: Mm,
33        title_ascender: Mm,
34        label_ascender: Mm,
35        text_ascender: Mm
36    ) -> Self {
37        let layout = BillLayout {
38            bill_data,
39            config: BillLayoutConfig {
40                has_acceptance_point: false,
41                max_height: PAYMENT_PART_MAX_HEIGHT,
42                debtor_box_height: DEBTOR_BOX_HEIGHT,
43                amount_section_top: PP_AMOUNT_SECTION_TOP,
44            },
45            language,
46            label_font_size,
47            text_font_size,
48            line_spacing,
49            extra_spacing,
50            title_ascender,
51            label_ascender,
52            text_ascender,
53        };
54        Self {
55            layout,
56            blocks: vec![
57                Box::new(TitleBlock { label: crate::LabelKey::PaymentPart }),
58                Box::new(InformationBlock {part: SlipPart::PaymentPart, offset: PP_INFO_SECTION_HORI_OFFSET, payable_box_width: DEBTOR_BOX_WIDTH_PP, payable_box_height: DEBTOR_BOX_HEIGHT}),
59                Box::new( QrBlock ),
60                Box::new(SpacerBlock{min_height: Mm(260f32)}),
61                Box::new(AmountBlock{part: SlipPart::PaymentPart, amount_box_width: AMOUNT_BOX_WIDTH_PP, amount_box_height: AMOUNT_BOX_HEIGHT_PP}),
62            ] }
63
64    }
65
66    pub fn render(&mut self, ops: &mut Vec<DrawOp>) {
67
68       // self.layout.compute_spacing();
69
70        let base_x = RECEIPT_WIDTH + MARGIN;
71
72        let mut left_cursor = ColumnCursor::new(
73            base_x,
74            A4_PAGE_HEIGHT - Mm(100f32),
75        );
76
77        let mut right_cursor = ColumnCursor::new(
78            base_x + PP_INFO_SECTION_HORI_OFFSET,
79            A4_PAGE_HEIGHT - Mm(100f32) + self.layout.label_ascender,
80        );
81
82        for block in &self.blocks {
83            match block.column() {
84                Column::Left => block.render(&mut self.layout, ops, &mut left_cursor),
85                Column::Right => block.render(&mut self.layout, ops, &mut right_cursor),
86                Column::Absolute => block.render(&mut self.layout, ops, &mut left_cursor),
87            }
88        }
89
90    }
91}
