18/// Trait for layout font metric strategies.
19pub trait LayoutStrategy<T: FontMetrics> {
20
21    const LABEL_SIZE: Pt;
22    const TEXT_SIZE: Pt;
23    const TITLE_SIZE: Pt;
24    const MAX_HEIGHT: Mm;
25
26    fn render(&mut self,
27              bill_data: &BillData,
28              language: Language,
29              metrics: &T,
30              ops: &mut Vec<DrawOp>
31    );
32
33}

43/// Base layout information shared by Payment Part and Receipt Part
44pub struct RenderContext<'a, T: FontMetrics> {
45    pub bill_data: &'a BillData,
46    pub language: Language,
47    pub metrics: &'a T,
48    pub label_size: Pt,
49    pub text_size: Pt,
50    pub title_size: Pt,
51    pub line_spacing: Mm,
52    pub label_ascender: Mm,
53    pub title_ascender: Mm,
54    pub text_ascender: Mm,
55    pub extra_spacing: Mm, // Calculated by your spacing logic
56}
57
58impl<'a, T: FontMetrics> RenderContext<'a, T> {
59    pub fn for_strategy<S: LayoutStrategy<T>>(
60        bill_data: &'a BillData,
61        language: Language,
62        metrics: &'a T
63    ) -> Self {
64        Self {
65            bill_data,
66            language,
67            metrics,
68            label_size: S::LABEL_SIZE,
69            text_size: S::TEXT_SIZE,
70            title_size: S::TITLE_SIZE,
71            line_spacing: metrics.line_height_mm(FontStyle::Regular, S::TEXT_SIZE),
72            label_ascender: metrics.ascender_mm(FontStyle::Bold, S::LABEL_SIZE),
73            text_ascender: metrics.ascender_mm(FontStyle::Regular, S::TEXT_SIZE),
74            title_ascender: metrics.ascender_mm(FontStyle::Bold, S::TITLE_SIZE),
75            extra_spacing: Mm(0.0), // Logic for Swiss spacing goes here
76        }
77    }
78}
