6use pdf_writer::Content;
7use qrcodegen::QrCode;
8use svg::Document;
9use svg::node::element::{Rectangle, Polygon as SvgPolygon};
10use crate::{CROSS_POLYGONS, CROSS_RECTS, PT_PER_MM};

12const QR_MM: f64 = 46.0;
13const CROSS_MM: f64 = 7.0;
14
15// Spec is a little misleading it talks about a quiet
16// zone of 4 modules (1.6 mm) but extends it to 5 mm.
17const QUIET_MM: f64 = 0f64;
18const CROSS_VIEWBOX: (f64, f64) = (19.8, 19.8);
19
20
21pub fn render_qr_pdf(content: &mut Content, qr: &QrCode, x_off: f32, y_off: f32) {
22
23        let modules = qr.size();
24        let drawable_pt = (QR_MM - 2.0 * QUIET_MM) as f32 * PT_PER_MM;
25        let module_pt = drawable_pt / modules as f32;
26        let quiet_pt = QUIET_MM as f32 * PT_PER_MM;
27
28        content.save_state();
29        content.set_fill_rgb(0.0, 0.0, 0.0); // Black modules
30
31        for row in 0..modules {
32            for col in 0..modules {
33                if qr.get_module(col, row) {
34                    let x = x_off + quiet_pt + (col as f32 * module_pt);
35                    // Remember PDF is Y-up!
36                    let y = y_off + quiet_pt + ((modules - 1 - row) as f32 * module_pt);
37
38                    content.rect(x, y, module_pt, module_pt);
39                }
40            }
41        }
42        content.fill_nonzero(); // Draw all modules at once for efficiency
43        content.restore_state();
44
45        content.save_state();
46        content.set_fill_rgb(1.0, 1.0, 1.0);
47        // Center of 46mm is 23mm. 7x7mm box starts at 23 - 3.5 = 19.5mm
48        let cross_box_offset = 19.5 * PT_PER_MM;
49        content.rect(x_off + cross_box_offset, y_off + cross_box_offset, 7.0 * PT_PER_MM, 7.0 * PT_PER_MM);
50        content.fill_nonzero();
51        content.restore_state();
52
53        // Draw the Swiss Cross (Vector)
54        draw_swiss_cross_vector(content, (x_off + (23.0 * PT_PER_MM)) as f64, (y_off + (23.0 * PT_PER_MM)) as f64);
55}
56
57pub fn draw_swiss_cross_vector(content: &mut Content, x_center: f64, y_center: f64) {
58    let cross_size_pt = 7.0 * PT_PER_MM as f64;
59    let scale = cross_size_pt / CROSS_VIEWBOX.0;
60
61    // Bottom-left corner of the 7x7mm cross area
62    let x_origin = x_center - (cross_size_pt / 2.0);
63    let y_origin = y_center - (cross_size_pt / 2.0);
64
65    content.save_state();
66
67    // 1. Black Background (The Polygon with rounded-ish corners)
68    content.set_fill_rgb(0.0, 0.0, 0.0);
69    for poly in CROSS_POLYGONS {
70        if let Some((first_x, first_y)) = poly.points.first() {
71            content.move_to((x_origin + (first_x * scale)) as f32, (y_origin + (cross_size_pt - (first_y * scale))) as f32);
72            for (px, py) in poly.points.iter().skip(1) {
73                content.line_to((x_origin + (px * scale)) as f32, (y_origin + (cross_size_pt - (py * scale))) as f32);
74            }
75            content.close_path();
76        }
77    }
78    content.fill_nonzero();
79
80    // 2. White Cross Arms (The Rects)
81    content.set_fill_rgb(1.0, 1.0, 1.0);
82    for r in CROSS_RECTS {
83        let rx = x_origin + (r.x * scale);
84        // Flip Y: SVG Y is distance from top, PDF Y is distance from bottom
85        let ry = y_origin + (cross_size_pt - ((r.y + r.height) * scale));
86        content.rect(rx as f32, ry as f32, r.width as f32 * scale as f32, r.height as f32 * scale as f32);
87    }
88    content.fill_nonzero();
89
90    content.restore_state();
91}

95/// Render a Swiss QR Code SVG (modules only).
96/// The Swiss cross area is cleared here.
97/// The official SVG cross is rendered later on top.
98pub fn render_qr_svg(qr: QrCode) -> Document {
99    let modules = qr.size();
100    let drawable_mm = QR_MM - 2.0 * QUIET_MM;
101    let module_mm = drawable_mm / modules as f64;
102
103    let mut doc = Document::new()
104        .set("width", format!("{}mm", QR_MM as usize))
105        .set("height", format!("{}mm", QR_MM as usize))
106        .set("viewBox", format!("0 0 {} {}", QR_MM, QR_MM));
107
108    for row in 0..modules {
109        for col in 0..modules {
110            if !qr.get_module(col, row) {
111                continue;
112            }
113
114            let x = QUIET_MM + col as f64 * module_mm;
115            let y = QUIET_MM + row as f64 * module_mm;
116
117            doc = doc.add(
118                Rectangle::new()
119                    .set("x", x)
120                    .set("y", y)
121                    .set("width", module_mm)
122                    .set("height", module_mm)
123                    .set("fill", "black"),
124            );
125        }
126    }
127
128    doc
129}

131/// Clears QR modules and draws Swiss cross centered
132pub fn add_swiss_cross(mut doc: svg::Document) -> svg::Document {
133    let scale = CROSS_MM / CROSS_VIEWBOX.0; // 7 / 19.8
134    let offset = (QR_MM - CROSS_MM) / 2.0;  // 19.5
135
136    // Clear QR modules underneath (white background)
137    doc = doc.add(
138        Rectangle::new()
139            .set("x", offset)
140            .set("y", offset)
141            .set("width", CROSS_MM)
142            .set("height", CROSS_MM)
143            .set("fill", "white"),
144    );
145
146    // Draw polygon parts of cross
147    for poly in CROSS_POLYGONS {
148        let points = poly
149            .points
150            .iter()
151            .map(|(x, y)| {
152                format!(
153                    "{},{}",
154                    x * scale + offset,
155                    y * scale + offset
156                )
157            })
158            .collect::<Vec<_>>()
159            .join(" ");
160
161        doc = doc.add(
162            SvgPolygon::new()
163                .set("points", points)
164                .set("fill", "black"),
165        );
166    }
167
168    for r in CROSS_RECTS {
169        doc = doc.add(
170            Rectangle::new()
171                .set("x", r.x * scale + offset)
172                .set("y", r.y * scale + offset)
173                .set("width", r.width * scale)
174                .set("height", r.height * scale)
175                .set("fill", "white"),
176        );
177    }
178
179    doc
180}
