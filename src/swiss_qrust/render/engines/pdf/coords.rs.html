6use crate::constants::A4_PAGE_HEIGHT;
7use crate::Mm;
8
9///
10/// Flips y coordinate. In the TV age, x,y started top left.
11/// Then, Apple and hence PostScript and PDF started using
12/// the origin at the conventional position bottom left.
13///
14/// This flips y so the layout layer can stay regardless of what the renderer does.
15/// So this converts from top down to a bottom up position
16///
17/// Distance from top of the layout area.
18
19#[derive(Copy, Clone, Debug)]
20pub struct LayoutY(pub Mm);
21
22impl LayoutY {
23
24    #[inline]
25    pub fn to_pdf(self) -> PdfY {
26        PdfY(A4_PAGE_HEIGHT - self.0)
27    }
28
29    pub fn to_pdf_with_height(self, page_height: Mm, obj_height: Mm) -> Mm {
30        page_height - self.0 - obj_height
31    }
32}
33
34#[derive(Copy, Clone, Debug)]
35pub struct PdfY(pub Mm);
