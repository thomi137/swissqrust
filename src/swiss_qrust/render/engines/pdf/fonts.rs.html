14const SYSTEM_INFO: SystemInfo = SystemInfo {
15    registry: Str(b"Adobe"),
16    ordering: Str(b"Identity"),
17    supplement: 0,
18};
19
20pub struct EmbeddedFont {
21    pub type0_ref: Ref,
22    pub face: Face<'static>, // Store this so we can encode/measure later
23}
24
25impl EmbeddedFont {
26    /// Maps a string to 2-byte Big-Endian Glyph IDs for pdf-writer
27    pub fn encode(&self, text: &str) -> Vec<u8> {
28        let mut bytes = Vec::with_capacity(text.len() * 2);
29        for c in text.chars() {
30            let gid = self.face.glyph_index(c).map(|g| g.0).unwrap_or(0);
31            bytes.push((gid >> 8) as u8);
32            bytes.push((gid & 0xFF) as u8);
33        }
34        bytes
35    }
36
37    /// Measures text width in PDF points for a given font size
38    pub fn measure(&self, text: &str, size: f32) -> f32 {
39        let mut width_units = 0.0;
40        for c in text.chars() {
41            let gid = self.face.glyph_index(c).unwrap_or(GlyphId(0));
42            width_units += self.face.glyph_hor_advance(gid).unwrap_or(0) as f32;
43        }
44        // (Width in units * size in pts) / UnitsPerEm
45        let width_pt = (width_units * size) / self.face.units_per_em() as f32;
46
47        width_pt * MM_PER_PT
48    }
49}

53pub fn embed_ttf_font(pdf: &mut Pdf, next_id: &mut Ref, custom_font_name: Name, font_data: &'static [u8]) -> EmbeddedFont {
54
55    let face = Face::parse(font_data, 0).expect("Invalid font data");
56    // Allocate IDs
57    let type0_ref = next_id.bump();
58    let cid_ref = next_id.bump();
59    let desc_ref = next_id.bump();
60    let stream_ref = next_id.bump();
61    let cmap_ref = next_id.bump();
62
63    // Extract unicode cmap from .ttf
64    let mut cmap = UnicodeCmap::<u16>::new(Name(b"Custom"), SYSTEM_INFO);
65    for subtable in face
66        .tables()
67        .cmap
68        .iter()
69        .flat_map(|c| c.subtables) {
70            if subtable.is_unicode() {
71                subtable.codepoints(|cp| {
72                    if let (Some(c), Some(gid)) = (char::from_u32(cp), subtable.glyph_index(cp)) {
73                        cmap.pair(gid.0, c);
74                    }
75                });
76                break;
77            }
78    }
79    pdf.stream(cmap_ref, &cmap.finish());
80
81    let units_per_em = face.units_per_em();
82
83    // Widths
84    // PDF CIDFont widths are: [first_gid [w1 w2 w3]]
85    let mut widths = Vec::new();
86    for gid in 0..face.number_of_glyphs() {
87        let width = face.glyph_hor_advance(GlyphId(gid)).unwrap_or(0);
88        let pdf_width = (width as f32 * 1000.0) / units_per_em as f32;
89        widths.push(pdf_width.round());
90    }
91
92
93    let compressed = compress_to_vec_zlib(font_data, 6);
94    let mut stream: Stream = pdf.stream(stream_ref, &compressed);
95    stream.pair(Name(b"Length1"), font_data.len() as i32);
96    stream.filter(Filter::FlateDecode);
97    stream.finish();
98
99    // Font Descriptor
100    // 1. Font Descriptor (Metadata)
101    let mut desc = pdf.font_descriptor(desc_ref);
102    let bbox = face.global_bounding_box();
103    desc.name(custom_font_name);
104    desc.bbox(pdf_writer::Rect::new(bbox.x_min as f32, bbox.y_min as f32, bbox.x_max as f32, bbox.y_max as f32));
105    desc.ascent(face.ascender() as f32);
106    desc.descent(face.descender() as f32);
107    desc.cap_height(face.capital_height().unwrap_or(700) as f32);
108    desc.italic_angle(face.italic_angle());
109    desc.flags(FontFlags::SYMBOLIC); // Adjust based on face.is_monospaced(), etc.
110    desc.stem_v(80.0);
111    if let Some(cap) = face.capital_height() {
112        desc.cap_height(cap as f32);
113    } else {
114        desc.cap_height(700.0); // Standard fallback
115    }
116    desc.font_file2(stream_ref);
117    desc.finish();
118
119    // CID Font
120    let mut cid = pdf.cid_font(cid_ref);
121    cid.subtype(CidFontType::Type2);
122    cid.base_font(custom_font_name);
123    cid.system_info(SYSTEM_INFO);
124    cid.font_descriptor(desc_ref);
125    // Add the Identity mapping manually here
126    cid.pair(Name(b"CIDToGIDMap"), Name(b"Identity"));
127
128    let mut w = cid.widths();
129    w.consecutive(0, widths);
130    w.finish();
131    cid.finish();
132
133    let mut type0 = pdf.type0_font(type0_ref);
134    type0.base_font(custom_font_name);
135    type0.encoding_predefined(Name(b"Identity-H"));
136    type0.descendant_font(cid_ref);
137    type0.to_unicode(cmap_ref);
138    type0.finish();
139
140    EmbeddedFont {
141        type0_ref,
142        face,
143    }
144}

146pub fn name(style: FontStyle) -> Name<'static> {
147    match style {
148        FontStyle::Regular => LIBERATION_SANS_REG_NAME,
149        FontStyle::Bold => LIBERATION_SANS_BOLD_NAME,
150    }
