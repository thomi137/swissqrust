7use std::fmt::{Display, Formatter};
8use serde::{Deserialize, Serialize};
9
10use crate::support::validators::*;
11
12#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
13#[serde(rename_all = "SCREAMING_SNAKE_CASE")]
14pub enum ReferenceType {
15    #[serde(rename = "NON")]
16    NoRef,
17
18    #[serde(rename = "QRR")]
19    QrRef(String),
20
21    #[serde(rename = "SCOR")]
22    Creditor(String),
23}
24
25/// There are three reference types:
26///  * None - A normal IBAN should not have a QR reference
27///  * QRR - If a QR IBAN is given, there MUST be a QR Reference
28///  * SCOR - If there is a normal IBAN, a SCOR or creditor IBAN can be added.
29impl ReferenceType {
30    //! Returns a reference type based on the given value.
31    //!
32    //! # Arguments
33    //!
34    //! * `value` - The value to be parsed.
35    //!
36    //! # Errors
37    //! 
38    //! * Returns `ReferenceError::InvalidReference` if the value is not a valid reference.
39    //! * Returns `ReferenceError::InvalidQrReference` if the value is a QR reference, but not valid.
40    //! * Returns `ReferenceError::InvalidIso11649Reference` if the value is a creditor reference, but not valid.
41    //!
42    //! # Examples
43    //!
44    //! ```
45    //! use swiss_qrust::ReferenceType;
46    //! let ref_type = ReferenceType::infer("RF18539007547034").unwrap();
47    //! assert_eq!(ref_type, ReferenceType::Creditor("RF18539007547034".into()));
48    //!```
49    //! Returns an error if the value is not a valid reference.
50    pub fn infer(value: &str) -> Result<Self, ReferenceError> {
51
52        // If value is empty, then Reference Type is none
53        if value.is_empty() {
54            return Ok(Self::NoRef);
55        }
56
57        // If value has exactly 27 numeric digits
58        if value.len() == 27usize && value.chars().all(|c| c.is_ascii_digit()) {
59            is_valid_qr_reference(value)?;
60            return Ok(Self::QrRef(value.to_owned()));
61        }
62
63        // ISO 11649 creditor reference: RF... + MOD97
64        if value.starts_with("RF") {
65            is_valid_iso11649_reference(value)?;
66            return Ok(Self::Creditor(value.to_owned()));
67        }
68
69        Err(ReferenceError::InvalidReference)
70    }
71
72    pub fn code(&self) -> &'static str {
73        match self {
74            ReferenceType::NoRef => "NON",
75            ReferenceType::QrRef(_) => "QRR",
76            ReferenceType::Creditor(_) => "SCOR",
77        }
78    }
79    pub fn requires_references(&self) -> bool {
80        matches!(self, ReferenceType::QrRef(_) | ReferenceType::Creditor(_))
81    }
82
83}
84
85impl Display for ReferenceType {
86    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
87        match self {
88            ReferenceType::NoRef => f.write_str("NON"),
89            ReferenceType::QrRef(reference) => write!(f,"QRR: {}",reference),
90            ReferenceType::Creditor(reference) => write!(f,"SCOR: {}",reference),
91        }
92    }
93}

95#[cfg(test)]
96mod tests {
97
98    use super::*;
99
100    #[test]
101    fn test_get_non_reference() {
102        let value = "";
103        let ref_type = ReferenceType::infer(value).unwrap();
104        assert_eq!(ref_type, ReferenceType::NoRef);
105    }
106
107    #[test]
108    fn test_get_qr_reference() {
109        let value = "210000000003139471430009017";
110        let ref_type = ReferenceType::infer(value).unwrap();
111        assert_eq!(ref_type, ReferenceType::QrRef(value.to_owned()));
112    }
113
114    #[test]
115    fn test_iso11649_reference() {
116        let value = "RF18539007547034";
117        let ref_type = ReferenceType::infer(value).unwrap();
118        assert_eq!(ref_type, ReferenceType::Creditor(value.to_owned()));
119    }
120
121    #[test]
122    fn reference_type_codes() {
123        assert_eq!(ReferenceType::NoRef.code(), "NON");
124        assert_eq!(ReferenceType::QrRef("210000000003139471430009017".into()).code(), "QRR");
125        assert_eq!(ReferenceType::Creditor("RF18539007547034".into()).code(), "SCOR");
126    }
