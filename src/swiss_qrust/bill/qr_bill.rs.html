12#[derive(Debug, Error, Clone, Eq, PartialEq)]
13pub enum QRBillError {
14    #[error("QR encoding failed")]
15    QrEncodingFailed,
16}

18/// According to the [spec] (https://www.six-group.com/dam/download/banking-services/standardization/qr-bill/ig-qr-bill-v2.3-de.pdf)
19/// section 4.1.4, there are two allowed line separators.
20#[derive(Debug, Default, Clone, PartialEq, Eq)]
21pub enum LineSeparator{
22    // Carriage return and line feed. Unicode U+000D and U+000A.
23    #[default]
24    CrLF,
25    // Line feed. Unicode U+000A.
26    Lf
27}
28impl LineSeparator {
29    pub fn as_str(&self) -> &'static str {
30        match self {
31            LineSeparator::CrLF => "\r\n",
32            LineSeparator::Lf => "\n",
33        }
34    }
35}
36impl Display for LineSeparator {
37    fn fmt(&self, f: &mut std::fmt::Formatter) -> std::fmt::Result {
38        match self {
39            LineSeparator::CrLF => f.write_str("\r\n"),
40            LineSeparator::Lf => f.write_str("\n"),
41        }
42    }
43}

45/// At the moment, only version "0200" is accepted. Starting from
46/// version 3, minors are accepted.
47///
48/// ```
49/// use swiss_qrust::qr_bill::Version;
50/// let version = Version{ major: 2, minor: 0};
51/// assert_eq!(version.qr_code_version(), "0200");
52/// ```
53///
54/// Alternatively, you can also use .to_string():
55///
56/// ```
57/// use swiss_qrust::qr_bill::Version;
58/// let version = Version{ major: 2, minor: 0};
59/// assert_eq!(version.to_string(), "0200");
60/// ```
61///
62/// Seems kind of redundant, but could be helpful internally to
63/// implement Display.
64#[derive(Debug, Clone, Copy, PartialEq, Eq)]
65pub struct Version {
66    pub major: u8,
67    pub minor: u8,
68}
69impl Version {
70    pub fn qr_code_version(&self) -> String {
71        format!("{:02}{:02}", self.major, self.minor)
72    }
73}
74impl Display for Version {
75    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
76        f.write_str(&self.qr_code_version())
77    }
78}

80const QR_TYPE: &str = "SPC";
81const VERSION: Version = Version { major: 2, minor: 0 };
82const CODING_TYPE: &str = "1";
83const TRAILER_EPD: &str = "EPD";

85pub struct QrBill<'a> {
86    pub bill_data: &'a BillData,
87    qr_type: String,
88    version: String,
89    coding_type: String,
90}
91impl<'a> QrBill
92<'a> {
93    pub fn new(
94        bill_data: &'a BillData,
95    ) -> Result<Self, QRBillError> {
96        let qr_type = QR_TYPE.to_string();
97        let version = VERSION.to_string();
98        let coding_type = CODING_TYPE.to_string();
99        Ok(QrBill{
100            bill_data,
101            qr_type,
102            version,
103            coding_type,
104        })
105    }
106
107    
108    pub fn create_qr_text(&self) -> Result<String, QRBillError> {
109
110        // TODO Make available for other line separator as well
111        let mut qr_text = QRTextBuilder::new(LineSeparator::default());
112
113        // Header Data. Not expected to change anytime soon.
114        qr_text.append_header(&self.qr_type, &self.version, &self.coding_type);
115
116        // IBAN - Mandatory
117        qr_text.append_data_field(Some(self.bill_data.iban.as_str()));
118        // Cdtr - Mandatory
119        qr_text.append_person(Some(&self.bill_data.creditor_address));
120        // UltmtCdtr - Has to be there, has to be empty
121        qr_text.append_person(None);
122
123        qr_text.append_data_field(self.bill_data.amount.as_deref());
124        qr_text.append_data_field(Some(&self.bill_data.currency.to_string()));
125
126        // UltmtDbtr - Debtor
127        qr_text.append_person(self.bill_data.debtor_address.as_ref());
128
129        let ref_type = &self.bill_data.reference_type;
130        // RmtInf + Tp - Reference Type
131        qr_text.append_data_field(Some(ref_type.code()));
132
133        // Reference
134        match ref_type {
135            ReferenceType::NoRef => qr_text.append_data_field(None),
136            ReferenceType::QrRef(reference) => qr_text.append_data_field(Some(reference)),
137            ReferenceType::Creditor(reference) => qr_text.append_data_field(Some(reference)),
138        }
139
140
141        // Additional information
142        qr_text.append_data_field(self.bill_data.unstructured_message.as_deref());
143
144        //Trailer - End of Payment Data
145        qr_text.append_data_field(Some(TRAILER_EPD));
146
147        // BIll Information
148        if let Some(billinfo) = &self.bill_data.bill_information {
149            qr_text.append_data_field(Some(billinfo));
150        }
151
152        // Alternative Schemes
153        for scheme in &self.bill_data.alternative_schemes {
154            if let Some(scheme) = scheme {
155                qr_text.append_data_field(Some(scheme));
156            }
157        }
158
159        Ok(qr_text.build())
160    }
161}

163pub struct QRTextBuilder {
164    text: String,
165    separator: LineSeparator,
166}
167impl QRTextBuilder {
168    pub fn new(separator: LineSeparator) -> Self {
169        Self {
170            text: String::new(),
171            separator
172        }
173    }
174
175    pub fn append_data_field(&mut self, value: Option<&str>) {
176        let value = value.unwrap_or("");
177        self.text.push_str(self.separator.as_str());
178        self.text.push_str(value);
179    }
180
181    pub fn append_person(&mut self, address: Option<&Address>) {
182        if let Some(addr) = address {
183            self.append_data_field(Some(&addr.address_type));
184            self.append_data_field(Some(&addr.name));
185            self.append_data_field(addr.street.as_deref());
186            self.append_data_field(addr.house_num.as_deref());
187            self.append_data_field(Some(&addr.plz));
188            self.append_data_field(Some(&addr.city));
189            self.append_data_field(Some(&addr.country.meta().alpha2));
190        } else {
191            for _ in 0..7 {
192                self.append_data_field(None);
193            }
194        }
195    }
196
197    pub fn append_header(&mut self, qr_type: &str, version: &str, coding_type: &str) {
198        self.text.push_str(qr_type);
199        self.append_data_field(Some(version));
200        self.append_data_field(Some(coding_type));
201    }
202
203    pub fn build(self) -> String {
204        self.text
205    }
206}

208pub fn encode_text_to_qr_code(qr_text: &str) -> Result<QrCode, QRBillError> {
209    QrCode::encode_text(qr_text, QrCodeEcc::Medium)
210        .map_err(|_| QRBillError::QrEncodingFailed)
211}
212
213pub fn qr_code(bill: &BillData) -> Result<QrCode, QRBillError> {
214    QrBill::new(bill)
215        .and_then(|b| b.create_qr_text())
216        .and_then(|txt| encode_text_to_qr_code(&txt))
217}
